const TOPICS = [...new Set(ALL_QUESTIONS.map(q => q.topic))].sort();
let selectedTopics = new Set(TOPICS);
let selectedNum = 5;
let quizQuestions = [];
let currentIdx = 0;
let score = 0;
let answered = false;

const tg = document.getElementById('topic-grid');
TOPICS.forEach(t => {
  const chip = document.createElement('button');
  chip.className = 'topic-chip active';
  chip.textContent = t;
  chip.onclick = () => toggleTopic(t, chip);
  tg.appendChild(chip);
});

function toggleTopic(topic, el) {
  if (selectedTopics.has(topic)) {
    selectedTopics.delete(topic);
    el.classList.remove('active');
  } else {
    selectedTopics.add(topic);
    el.classList.add('active');
  }
}

function selectNum(n) {
  selectedNum = n;
  document.querySelectorAll('.num-btn').forEach(b => b.classList.remove('selected'));
  event.target.classList.add('selected');
  document.getElementById('custom-num').value = '';
}

function selectCustom(v) {
  document.querySelectorAll('.num-btn').forEach(b => b.classList.remove('selected'));
  const n = parseInt(v);
  if (n >= 1 && n <= 50) selectedNum = n;
}

function shuffle(arr) {
  const a = [...arr];
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}

function startQuiz() {
  const pool = ALL_QUESTIONS.filter(q => selectedTopics.has(q.topic));
  if (pool.length === 0) { alert('Please select at least one topic!'); return; }
  const count = Math.min(selectedNum, pool.length);
  quizQuestions = shuffle(pool).slice(0, count);
  currentIdx = 0;
  score = 0;
  answered = false;
  document.getElementById('setup-screen').style.display = 'none';
  document.getElementById('quiz-screen').style.display = 'block';
  document.getElementById('q-total').textContent = quizQuestions.length;
  loadQuestion();
}

function loadQuestion() {
  answered = false;
  const q = quizQuestions[currentIdx];
  document.getElementById('q-num').textContent = currentIdx + 1;
  document.getElementById('score-live').textContent = score;
  document.getElementById('q-topic').textContent = q.topic;
  document.getElementById('q-text').textContent = q.q;

  const pct = (currentIdx / quizQuestions.length) * 100;
  document.getElementById('progress-fill').style.width = pct + '%';

  const choicesEl = document.getElementById('choices');
  choicesEl.innerHTML = '';
  const letters = ['A','B','C','D'];
  q.choices.forEach((c, i) => {
    const btn = document.createElement('button');
    btn.className = 'choice-btn';
    btn.innerHTML = `<span class="choice-letter">${letters[i]}</span><span>${c}</span>`;
    btn.onclick = () => answer(i);
    choicesEl.appendChild(btn);
  });

  const fb = document.getElementById('feedback');
  fb.className = 'feedback';
  document.getElementById('next-btn').className = 'next-btn';
}

function answer(idx) {
  if (answered) return;
  answered = true;

  const q = quizQuestions[currentIdx];
  const buttons = document.querySelectorAll('.choice-btn');
  buttons.forEach(b => b.disabled = true);

  const isCorrect = idx === q.correct;
  if (isCorrect) {
    score++;
    document.getElementById('score-live').textContent = score;
    buttons[idx].classList.add('correct');
  } else {
    buttons[idx].classList.add('wrong');
    buttons[q.correct].classList.add('correct');
  }

  const fb = document.getElementById('feedback');
  fb.className = 'feedback show ' + (isCorrect ? 'correct-fb' : 'wrong-fb');

  document.getElementById('fb-icon').textContent = isCorrect ? '✅' : '❌';
  document.getElementById('fb-title').textContent = isCorrect ? 'Correct!' : 'Not quite…';

  const cl = document.getElementById('correct-line');
  if (!isCorrect) {
    cl.style.display = 'block';
    cl.textContent = `✓ Correct answer: ${String.fromCharCode(65 + q.correct)}. ${q.choices[q.correct]}`;
  } else {
    cl.style.display = 'none';
  }

  document.getElementById('fb-explanation').innerHTML = `<strong>Explanation:</strong> ${q.explanation}`;

  const nb = document.getElementById('next-btn');
  nb.className = 'next-btn show';
  nb.textContent = currentIdx < quizQuestions.length - 1 ? 'Next Question →' : 'See Results →';
}

function nextQuestion() {
  currentIdx++;
  if (currentIdx >= quizQuestions.length) {
    showResults();
  } else {
    document.getElementById('quiz-screen').style.display = 'block';
    loadQuestion();
    window.scrollTo({ top: 0, behavior: 'smooth' });
  }
}

function showResults() {
  document.getElementById('quiz-screen').style.display = 'none';
  document.getElementById('results-screen').style.display = 'block';

  const total = quizQuestions.length;
  const pct = Math.round((score / total) * 100);

  document.getElementById('score-fraction').textContent = score + ' correct out of ' + total;
  document.getElementById('stat-correct').textContent = score;
  document.getElementById('stat-wrong').textContent = total - score;
  document.getElementById('stat-total').textContent = total;

  const pctEl = document.getElementById('score-pct');
  pctEl.textContent = pct + '%';

  let emoji, title, subtitle, cls;
  if (pct >= 90) { emoji='🏆'; title='Excellent Work!'; subtitle='You\'re ready for the Regents!'; cls='excellent'; }
  else if (pct >= 75) { emoji='🎉'; title='Great Job!'; subtitle='Keep practicing — you\'re almost there.'; cls='good'; }
  else if (pct >= 60) { emoji='📚'; title='Good Effort'; subtitle='Review the topics you missed and try again.'; cls='okay'; }
  else { emoji='💪'; title='Keep Studying!'; subtitle='Don\'t give up — chemistry takes practice!'; cls='poor'; }

  document.getElementById('results-emoji').textContent = emoji;
  document.getElementById('results-title').textContent = title;
  document.getElementById('results-subtitle').textContent = subtitle;
  pctEl.className = 'score-big ' + cls;

  window.scrollTo({ top: 0, behavior: 'smooth' });
}

function restart() {
  document.getElementById('results-screen').style.display = 'none';
  document.getElementById('setup-screen').style.display = 'block';
  window.scrollTo({ top: 0, behavior: 'smooth' });
}
